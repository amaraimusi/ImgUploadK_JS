JavaScript 
	<script>
	$(function(){
		var imgUploadK = new ImgUploadK();
		imgUploadK.extendDnD({
			'rap_slt':'#test1',
			'file_upload_slt':'#file1',
			'init_display_slt':'#test1_init',
			'img_preview_slt':'#img1',
		},function(files){
			$('#res').append(files[0].name);
		});
	});
	</script>
